      char path[128];
      int mode;
      struct node next;
      struct node       prev;
      }Node;

typedef struct list{
Node first;
Node last;
}List;

List init_link()//创建初始化链表
{
List list = (List*) malloc(sizeof(List));
if(list == NULL)
{
perror(“List malloc error”);
return NULL;
}
list->first = list->last = NULL;
return list;
}

List add_link(List list,char filename,int mode)
{
Node p = (Node *)malloc(sizeof(Node));
if(p==NULL)
{
perror(“node malloc error”);
return NULL;
}
p->mode=mode;
strcpy(p->path,filename);
p->next = p->prev = NULL;
if(list->first == NULL)
{
list->first = list->last = p; //头尾都是他
list->last->next = list->first; // 尾巴的下一个是 第一个
list->first->prev = list->last;
}
else
{
//尾插
list->last->next = p; // 原最后一个的下一个是新节点
p->prev = list->last;//
list->last = p; //新节点成为新尾巴
list->last->next = list->first; //尾巴 的下一个是 第一个
list->first->prev = list->last;
}

return list;

}

队列：
限定在表头删除 表尾插入的线性表 称之为 队列！
队头：指的是删除的那一端！
队尾：插入的一端
原则：先进先出 first in first out

栈：
限定在表的一端进行插入和删除的线性表（stack）!
栈顶：就是插入和删除的那一端！
栈低：不能插入和删除的一端
